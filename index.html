    <script>

        var lottoNumbers = [];


        function generateLottoNumbers() {

            var selectedNumbers = [];

            var checkboxes = document.getElementsByClassName('checkbox');


            for (var i = 0; i < checkboxes.length; i++) {

                if (checkboxes[i].checked) {

                    selectedNumbers.push(parseInt(checkboxes[i].value));

                }

            }


            var requiredSelections = 6 - selectedNumbers.length;

            var availableNumbers = [];

            for (var j = 1; j <= 45; j++) {

                if (!selectedNumbers.includes(j)) {

                    availableNumbers.push(j);

                }

            }


            var generatedNumbers = [];

            for (var k = 0; k < requiredSelections; k++) {

                var randomIndex = Math.floor(Math.random() * availableNumbers.length);

                var randomNumber = availableNumbers[randomIndex];

                generatedNumbers.push(randomNumber);

                availableNumbers.splice(randomIndex, 1);

            }


            var finalNumbers = selectedNumbers.concat(generatedNumbers);

            finalNumbers.sort(function(a, b) {

                return a - b;

            });


            var lottoResult = document.getElementById('lottoResult');

            var lottoList = document.getElementById('lottoList');


            lottoResult.innerText = finalNumbers.join(', ');


            var newLottoNumber = "[" + (lottoNumbers.length + 1) + "] " + finalNumbers.join(', ');

            lottoNumbers.push(newLottoNumber);

            lottoList.innerHTML = lottoNumbers.join('<br>');

        }

    </script>
